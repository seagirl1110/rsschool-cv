function twiceAsOld(dadYearsOld, sonYearsOld) {
    let direction = dadYearsOld / sonYearsOld;
    if (direction === 2) {
        return 0;
    } else if (direction > 2) {
        let yearsWill = 0;
        while (direction !== 2) {
            dadYearsOld++;
            sonYearsOld++;
            yearsWill++;
            direction = dadYearsOld / sonYearsOld;
        }
        return yearsWill;
    } else {
        let yearsAgo = 0;
        while (direction !== 2) {
            dadYearsOld--;
            sonYearsOld--;
            yearsAgo++;
            direction = dadYearsOld / sonYearsOld;
        }
        return yearsAgo;
    }
}
